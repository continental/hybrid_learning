[docs]    def __init__(self,
                 model: torch.nn.Module,
                 concepts: Collection[Concept],
                 objectives: Collection[torch.nn.Module],
                 layer_infos: Dict[str, Dict]
                 ):
        """Init.

        :param model: the DNN to embed concepts into
        :param concepts: the concepts that should be embedded
        :param objectives: the objectives to respect so far;
            objectives = loss function terms & associated training data/labels
        """
        # Value check: Are concept IDs unique?
        c_names_set = set()
        for c_name in (c for c in [c.name for c in concepts]
                       # select or add
                       if c in c_names_set or (not c_names_set.add(c))):
            raise ValueError("Duplicate concept name {}".format(c_name))

        self.model = model
        self.concepts: List[Concept] = list(concepts)  # order must be fixed!
        self.embeddings: List[ConceptEmbedding] = []  # embeddings of concepts
        self.objectives = objectives
        self.objective_weights = [1] * len(objectives)
        self.layer_infos: Dict[str, Dict] = layer_infos



[docs]    def embed_concepts(self, force_concept_analysis: bool = False):
        """Try to employ embeddings in the DNN and corresponding outputs for
        all concepts.

        :param force_concept_analysis: whether to force to determine the
            layer in which to embed the concept even concept analysis to do
            this was conducted before
        """
        c_names = [c.name for c in self.concepts
                   if c.name not in [o.name for o in self.objectives]] \
            if not force_concept_analysis else [c.name for c in self.concepts]
        # Embed all concepts
        for c_name in c_names:
            self.embed_concept(c_name)



[docs]    def embed_concept(self, concept_name: str):
        """Enforce one concept.

        :param concept_name: name of the concept to enforce.
        """
        if concept_name not in [c.name for c in self.concepts]:
            raise AttributeError("No concept of name {} is known."
                                 .format(concept_name))
        concept_idx = [c.name for c in self.concepts].index(concept_name)
        concept = self.concepts[concept_idx]

        # Get the layer in which the concept is currently best embedded
        analyser: analysis.ConceptAnalysis = analysis.ConceptAnalysis(
            concept, self.model, layer_infos=self.layer_infos)
        embedding: ConceptEmbedding = analyser.best_embedding()
        embedding_quality: pd.DataFrame = analyser.evaluate_embedding(embedding)
        self.update_concept_embedding(concept_name, embedding)
        # If the embedding quality is not sufficient, try to improve:
        if not concept.targets_met(embedding_quality.kpi_values):
            self.enforce_objective(concept_name)

[docs]    def verify_embedding_qualities(self):
        """Yield the truth value, whether all embeddings reached their
        concept KPI target values."""
        # Have all concepts been embedded already?
        if not all([c.name in [o.name for o in self.objectives]
                    for c in self.concepts]):
            print("WARNING: Verification was attempted before all concepts "
                  "were embedded!")
        return all([self.verify_embedding_quality(c.name)
                    for c in self.concepts])



[docs]    def verify_embedding_quality(self, concept_name):
        """Yield the truth value, whether the embedding reaches its concept
        KPI target value."""
        if concept_name not in [c.name for c in self.concepts]:
            raise AttributeError("No concept of name {} is known."
                                 .format(concept_name))
        concept_idx = [c.name for c in self.concepts].index(concept_name)
        concept = self.concepts[concept_idx]
        return concept.targets_met(self.get_kpi_values(concept_name))
